#include <Servo.h>

Servo solarServo;

int ldrLeft = A0;
int ldrRight = A1;

int servoPos = 90;

void setup() {
  solarServo.attach(9);

  solarServo.write(servoPos);

  Serial.begin(9600);
}

void loop() {

  int leftValue = analogRead(ldrLeft);
  int rightValue = analogRead(ldrRight);

  Serial.print("Left: ");
  Serial.print(leftValue);
  Serial.print(" Right: ");
  Serial.println(rightValue);

  int difference = leftValue - rightValue;
  Serial.print("Difference: ");
  Serial.println(difference);

  if (difference > 50) {
    servoPos++;
  }

  if (difference < -50) {
    servoPos--;
  }

  servoPos = constrain(servoPos, 0, 360);

  solarServo.write(servoPos);
  Serial.print("Serial Position: ");
  Serial.println(servoPos);

  delay(50);
}
